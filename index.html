async function fetchIndicadores(ticker) { const grid = document.getElementById('indicadores-grid'); grid.innerHTML = '
Buscando indicadores no mercado...
'; try { const res = await fetch(`${URL_BACKEND}/api/indicadores/${ticker}`); const data = await res.json(); // TRATAMENTO DE ERRO ELEGANTE if (data.status === "bloqueado") { grid.innerHTML = `
⚠️ O provedor de dados limitou os acessos temporariamente (IP sobrecarregado). Tente novamente mais tarde.
`; return; } if (Object.keys(data).length === 0) { grid.innerHTML = '
Indicadores não disponíveis para este ativo.
'; return; } const fmtNum = (num, sufixo='') => (num !== null && !isNaN(num)) ? parseFloat(num).toFixed(2) + sufixo : '-'; const fmtPct = (num) => (num !== null && !isNaN(num)) ? (parseFloat(num) * 100).toFixed(2) + '%' : '-'; grid.innerHTML = `
P/L${fmtNum(data.pl)}
P/VP${fmtNum(data.pvp)}
Div. Yield (D.Y)${fmtPct(data.dy)}
ROE${fmtPct(data.roe)}
ROA${fmtPct(data.roa)}
Margem Líquida${fmtPct(data.margemLiquida)}
Dív. Líquida / PL${fmtPct(data.dividaPL ? data.dividaPL / 100 : null)}
`; } catch(e) { grid.innerHTML = '
Erro de conexão ao buscar indicadores.
'; } }
